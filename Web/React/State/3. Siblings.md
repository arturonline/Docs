# Siblings

A child component updates its parentâ€™s state, and the parent passes that state to a sibling component.

```jsx
// App.js
import React from 'react';
import logo from './logo.svg';
import './App.css';
import { Parent } from './Parent';
import Child from './Child';

function App() {
  return (
    <div>
      <Parent />
    </div>
  );
}

export default App;
```

```jsx
// Parent.js
import React from 'react';
import Child from './Child';
import Sibling from './sibling';

export class Parent extends React.Component {
  constructor(props) {
    super(props);
    this.state = { value: 'guapa'}
  }

  onchangeName = () => {
    this.setState({
      value: 'guapa y bonica'
    });
  }
  render() {
    return <div>
            <p>Cari es {this.state.value}</p>
            <Child onClick={this.onchangeName} />
            <Sibling value={this.state.value} />
           </div>
  }
}
```

```jsx
// Child.js
import React from 'react';

class Child extends React.Component {
  constructor(props) {
    super(props);

    this.handleChange = this.handleChange.bind(this);
  }

  handleChange(e) {
    let value = e.target.value;
    this.props.onClick(value);
  }

  render() {
    return (
      <div>
        <button onClick={this.handleChange}>Press me!</button>
      </div> );
  }
}
export default Child;
```

// Sibling.js
import React from 'react';
